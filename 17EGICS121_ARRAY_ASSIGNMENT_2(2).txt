#include <stdio.h>
#include <string.h>
int main()
{
    int i,j,n,s;
    char temp;
    printf("\nenter no of chars: ");
    scanf("%d",&n);
    char a[n];
    printf("\nenter string: ");
    scanf("%s",&a);
    //it is assumed that characters will be equal to 'n' otherwise we can use 'for' loop.
    printf("\nenter no of right shift steps: ");
    scanf("%d",&s);
    if(s%n==0&&(strlen(a)==n)){
        printf("after shift: %s\n",a);
    }
    else{
        while(s/n != 0){
            printf("%d\n",s/n);
            s=s%n;
        }
        
        for(i=0;i<s;i++){
            temp = a[n-1];
            for (j=n-2; j>=0; j--){
                a[j+1] = a[j];
            }
            a[0] = temp;
        }
    if(s%n!=0){
        printf("\n\nresult: %s",a);
    }
}
}
